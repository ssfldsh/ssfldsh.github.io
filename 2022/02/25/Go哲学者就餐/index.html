使用Golang并发工具包解决哲学者就餐问题

1226. 哲学家进餐 - 力扣（LeetCode）

leetCode的并发编程题没有Go的测试环境，在本地自己写了个哲学者就餐问题，通过观察程序能否长时间保持运行状态而不死锁来判断程序逻辑是否正确。

package main

import (
    "fmt"
    "sync"
    "time"
)

func main() {
    // 初始变量1:筷子数
    philoNum := 5
    // 初始变量2:forkSate数组,表示每个筷子的状态
    // 5个哲学家对应5个线程，他们不断遵循一定规则修改筷子状态：
    // 规则一：拿到左右两双筷子就能吃饭。仅仅按照这个规则编程程序很容易发生死锁
    // 必须要引入其他规则才能使得5个线程没有死锁的情况出现，能一直运行下去 。
    // 这里采用只有同时拿到左右两只筷子才能吃饭(修改筷子状态，并调用eat方法)的规则来防止线程死锁。
    forkState := [5]int{0, 0, 0, 0, 0}
    mutex := sync.Mutex{}
    // channel数组:
    canRunPhilo := make([]chan int, 5)
    for i := 0; i < 5; i++ {
        canRunPhilo[i] = make(chan int, 1)
    }

    for i := 0; i < philoNum; i++ {
        // 匿名函数传参:
        go func(i int) {
            for {
        // 左右两只筷子状态都为0时:
                mutex.Lock()
                if forkState[i] == 0 && forkState[(i-1+philoNum)%philoNum] == 0 {
                    forkState[i] = 1
                    forkState[(i-1+philoNum)%philoNum] = 1
                    canRunPhilo[i] <- 0
                }
                mutex.Unlock()

        // 成功拿起两只筷子后：调用eat方法，输出5只筷子的状态:
                <-canRunPhilo[i]
                // eat(i)
                mutex.Lock()
                fmt.Print("筷子状态:")
                for i := 0; i < philoNum; i++ {
                    if forkState[i] != 0 {
                        fmt.Print("#", i)
                    }
                }
                fmt.Println()
                time.Sleep(100 * time.Microsecond)
                mutex.Unlock()

        // 就餐完毕后，执行放下筷子（即将筷子状态重新置为0）的操作:
                mutex.Lock()
        		// 放下左手筷子（即将左手筷子的状态置为0），同时判断左手边的哲学家（一个线程）是否拥有就餐资格:
                forkState[i] = 0
                if forkState[(i+1)%philoNum] == 0 {
                    forkState[i] = 1
                    forkState[(i+1)%philoNum] = 1
                    canRunPhilo[(i+1)%philoNum] <- 0
                }
        // 放下右手筷子，同时判断右手边的哲学家（一个线程）是否拥有就餐资格:
                forkState[(i-1+philoNum)%philoNum] = 0
                if forkState[(i-2+philoNum)%philoNum] == 0 {
                    forkState[(i-1+philoNum)%philoNum] = 1
                    forkState[(i-2+philoNum)%philoNum] = 1
                    canRunPhilo[(i-1+philoNum)%philoNum] <- 0
                }
                mutex.Unlock()
        // 哲学家（线程）i的任务执行完毕!
            }
        }(i)
    }
    // time.Sleep(10 * time.Second)
    // for {
    // 	time.Sleep(10 * time.Microsecond)
    // }
    select {}
}

func eat(i int) {
    fmt.Printf("now %d begin eating...\n", i)
}
// 总结：
// 通过 boolean forkStae数组的值 来判断 筷子是否在哲学家手中
// 通过 创建5个线程来模拟哲学家就餐过程
// 通过一定的策略来避免死锁
// 改写：题目很适合使用条件变量来代替channel控制线程同步
// 避免哲学家就餐（线程运行）时发生死锁还存在其他规则，比如每次只允许4个筷子被同时拿起。
