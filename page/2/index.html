使用Golang并发工具包解决哲学者就餐问题

1226. 哲学家进餐 - 力扣（LeetCode）

leetCode的并发编程题没有Go的测试环境，在本地自己写了个哲学者就餐问题，通过观察程序能否长时间保持运行状态而不死锁来判断程序逻辑是否正确。

package main

import (
    "fmt"
    "sync"
    "time"
)

func main() {
    // 初始变量1:筷子数
    philoNum := 5
    // 初始变量2:forkSate数组,表示每个筷子的状态
    // 5个哲学家对应5个线程，他们不断遵循一定规则修改筷子状态：
    // 规则一：拿到左右两双筷子就能吃饭。仅仅按照这个规则编程程序很容易发生死锁
    // 必须要引入其他规则才能使得5个线程没有死锁的情况出现，能一直运行下去 。
    // 这里采用只有同时拿到左右两只筷子才能吃饭(修改筷子状态，并调用eat方法)的规则来防止线程死锁。
    forkState := [5]int{0, 0, 0, 0, 0}
    mutex := sync.Mutex{}
    // channel数组:
    canRunPhilo := make([]chan int, 5)
    for i := 0; i < 5; i++ {
        canRunPhilo[i] = make(chan int, 1)
    }

    for i := 0; i < philoNum; i++ {
        // 匿名函数传参:
        go func(i int) {
            for {
        // 左右两只筷子状态都为0时:
                mutex.Lock()
                if forkState[i] == 0 && forkState[(i-1+philoNum)%philoNum] == 0 {
                    forkState[i] = 1
                    forkState[(i-1+philoNum)%philoNum] = 1
                    canRunPhilo[i] <- 0
                }
                mutex.Unlock()

        // 成功拿起两只筷子后：调用eat方法，输出5只筷子的状态:
                <-canRunPhilo[i]
                // eat(i)
                mutex.Lock()
                fmt.Print("筷子状态:")
                for i := 0; i < philoNum; i++ {
                    if forkState[i] != 0 {
                        fmt.Print("#", i)
                    }
                }
                fmt.Println()
                time.Sleep(100 * time.Microsecond)
                mutex.Unlock()

        // 就餐完毕后，执行放下筷子（即将筷子状态重新置为0）的操作:
                mutex.Lock()
        		// 放下左手筷子（即将左手筷子的状态置为0），同时判断左手边的哲学家（一个线程）是否拥有就餐资格:
                forkState[i] = 0
                if forkState[(i+1)%philoNum] == 0 {
                    forkState[i] = 1
                    forkState[(i+1)%philoNum] = 1
                    canRunPhilo[(i+1)%philoNum] <- 0
                }
        // 放下右手筷子，同时判断右手边的哲学家（一个线程）是否拥有就餐资格:
                forkState[(i-1+philoNum)%philoNum] = 0
                if forkState[(i-2+philoNum)%philoNum] == 0 {
                    forkState[(i-1+philoNum)%philoNum] = 1
                    forkState[(i-2+philoNum)%philoNum] = 1
                    canRunPhilo[(i-1+philoNum)%philoNum] <- 0
                }
                mutex.Unlock()
        // 哲学家（线程）i的任务执行完毕!
            }
        }(i)
    }
    // time.Sleep(10 * time.Second)
    // for {
    // 	time.Sleep(10 * time.Microsecond)
    // }
    select {}
}

func eat(i int) {
    fmt.Printf("now %d begin eating...\n", i)
}
// 总结：
// 通过 boolean forkStae数组的值 来判断 筷子是否在哲学家手中
// 通过 创建5个线程来模拟哲学家就餐过程
// 通过一定的策略来避免死锁
// 改写：题目很适合使用条件变量来代替channel控制线程同步
// 避免哲学家就餐（线程运行）时发生死锁还存在其他规则，比如每次只允许4个筷子被同时拿起。

Go并发：控制多个线程中部分功能的执行顺序

题目一：顺序打印

要求1：现在有三个线程A、B、C，它们的功能分别为打印A、B、C各五次，

要求修改代码：使用同步语句控制线程执行顺序，使得A的打印发生在B之前，B的打印发生在C之前

package main

import (
    "fmt"
    "sync"
    "time"
)

func main() {
    var wg sync.WaitGroup
    wg.Add(3)
    nowBThreadCanRun := make(chan int)
    nowCThreadCanRun := make(chan int)
    go func() {
        // ３、打印A不需要依赖于谁，直接打印即可
        for i := 0; i < 5; i++ {
            fmt.Println("A")
        }
        nowBThreadCanRun <- 1
        // A线程阻塞时，可以执行B线程，体现了多线程的意义
        time.Sleep(1000 * time.Millisecond)
        wg.Done()
    }()
    go func() {
        //　２、在A线程和B线程中使用使用nowBThreadCanRun，使得A线程中的打印操作发生在B线程之前
        <-nowBThreadCanRun
        for i := 0; i < 5; i++ {
            fmt.Println("B")
        }
        nowCThreadCanRun <- 2
        time.Sleep(1000 * time.Millisecond)
        wg.Done()
    }()
    go func() {
        //　２、在B线程和C线程中使用使用nowCThreadCanRun，使得B线程中的打印操作发生在C线程之前
        <-nowCThreadCanRun
        for i := 0; i < 5; i++ {
            fmt.Println("C")
        }
        wg.Done()
    }()
    time.Sleep(100 * time.Millisecond)
    fmt.Println("main thread end")
    // 使用waitGroup阻塞main线程，防止main线程过早执行结束
    wg.Wait()
}

// 不使用多线程也能实现顺序打印ABC的效果，但是因为在print过程中，加入了很多阻塞cpu的操作（如sleep）
// 使得以非并发方式实现的代码运行时间很长：至少大于代码中所有sleep的时间的总和：
// 以多线程方式实现的顺序打印ABC的代码运行时间为1.115s左右，比顺序执行要快，顺序执行的话，3个sleep操作就至少需要执行2.1s

题目二：现在有A、B两个线程，它们的功能分别为打印A、B各5次

现在要求修改代码，使得：

A和B交替打印：

具体规则：打印一个A后才能打印一个B；打印一个B后才能打印一个A;A先被打印。

package main

import (
    "fmt"
    "sync"
    "time"
)

// const声明以及赋值:
const looptime int = 100

func main() {
    var wg sync.WaitGroup
    wg.Add(2)
    start := time.Now()
    // 交替打印A、B，使用1个或者2个cahnnel来实现:
    canPrintFoo := make(chan int, 1)
    canPrintBar := make(chan int, 1)
    // 在A、B两个线程中引入：canPrintA channel使得对A的打印发生在B之前：
    // 在A、B两个线程中引入：canPrintB channel使得对B的打印发生在A之前：
    // 在创建A、B两个线程前先 canPrintA channel里注入数据。
    canPrintFoo <- 0
    go func() {
        for i := 0; i < looptime; i++ {
            <-canPrintFoo
            fmt.Print("foo")
            canPrintBar <- 0
        }
        defer wg.Done()
    }()
    go func() {
        for i := 0; i < looptime; i++ {
            <-canPrintBar
            fmt.Print("bar")
            canPrintFoo <- 0
        }
        defer wg.Done()
    }()
    wg.Wait()
    fmt.Println("the main function has been run:", time.Since(start))
}

多线程程序适合在I/O密集型应用中使用，当一个线程处于I/O状态时，CPU是空闲的，可以被用来执行其他线程。

大部分网络应用都是I/O密集型的，适合使用并发优化。

常见的I/O优化方式还有

同步非阻塞IO： 在一个线程里使用多路复用，同时监听多个连接。多个连接共享一个线程，避免大量线程创建带来的时间和空间开销。

异步非阻塞IO：不需要主动轮询，访问每个连接状态。当被调用方执行完毕式，主动通知调用方。

对于计算密集型任务，适合使用并行优化：

如游戏还有科学计算领域使用GPU并行处理矩阵计算、处理图像数据。

合适的数据结构和算法也能降低程序的复杂度。某些算法的优化空间特别大，如使用动态规划代替暴力搜索，能将某些O(2^n)的算法优化到O(n)。

Redis持久化

AOF和RDB的取舍

目前，Redis 的持久化主要有两大机制，即 AOF（Append Only File）日志和 RDB 快照。

AOF记录的是操作命令，而不是实际的数据。
用 AOF 方法进行故障恢复的时候，需要逐一把操作日志都执行一遍。如果操作日志非常多，Redis 就会恢复得很缓慢，影响到正常使用。这当然不是理想的结果。
使用AOF备份的特点是慢而全面，数据内容的丢失可以控制在秒级。

使用RDB快照在做数据恢复时比较快。
缺点是备份内容不够详细，数据粒度只能达到分钟级。
频繁的全量快照造成大量高频磁盘读写，fork开销也很大。

实际生产中，可以考虑两者的结合，如：

Redis 4.0 中提出了一个混合使用 AOF 日志和内存快照的方法。简单来说，内存快照以一定的频率执行，在两次快照之间，使用 AOF 日志记录这期间的所有命令操作。

这样数据恢复速度可以得到提升，能恢复出的数据也更加全面。
